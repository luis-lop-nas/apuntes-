\section{Tema 1: Oscilaciones Lineales}
% ══════════════════════════════════════════════════════════════════════════

\subsection{Introducción}

Para una partícula existe una posición de equilibrio estable que tomamos como origen de
coordenadas. Cuando la partícula se desplaza del equilibrio aparece una fuerza recuperadora
que tiende a devolverla a dicho origen.

\begin{definicion}[Fuerza recuperadora lineal]
Expandiendo $F(x)$ en serie de Taylor alrededor del equilibrio ($x = 0$):
\begin{equation}
  F(x) = F_0 + x\!\left(\frac{dF}{dx}\right)_{\!0}
         + \frac{1}{2}x^2\!\left(\frac{d^2F}{dx^2}\right)_{\!0} + \cdots
  \label{eq:taylor}
\end{equation}
En el equilibrio $F_0 = 0$. Para desplazamientos suficientemente pequeños se desprecia
todo término de orden cuadrático o superior, quedando la \textbf{ley de Hooke}:
\begin{mdframed}[style=resultado]
\begin{equation}
  F(x) = -kx, \qquad k \equiv -\!\left(\frac{dF}{dx}\right)_{\!0} > 0
  \label{eq:hooke}
\end{equation}
\end{mdframed}
donde $k\,[\mathrm{N\,m^{-1}}]$ es la \emph{constante elástica} o \emph{de rigidez}. La
condición $k > 0$ garantiza la estabilidad: la fuerza se opone al desplazamiento.
\end{definicion}

\subsection{Oscilador armónico simple}

\begin{definicion}[Oscilador armónico simple]
Un \emph{oscilador armónico simple} (OAS) es un sistema de masa $m$ sujeto a una fuerza
recuperadora lineal $F = -kx$.
\end{definicion}

\begin{proposicion}[Ecuación del oscilador armónico simple]
La ecuación de movimiento del OAS es:
\begin{mdframed}[style=resultado]
\begin{equation}
  \ddot{x} + \omega_0^2\,x = 0, \qquad \omega_0^2 \equiv \frac{k}{m}
  \label{eq:oas}
\end{equation}
\end{mdframed}
donde $\omega_0\,[\mathrm{rad\,s^{-1}}]$ es la \emph{frecuencia propia} o
\emph{pulsación natural} del oscilador.
\end{proposicion}

\begin{proof}
Aplicando la segunda ley de Newton con la fuerza de Hooke~\eqref{eq:hooke}:
\[
  m\ddot{x} = -kx \implies \ddot{x} + \frac{k}{m}\,x = 0.
\]
Identificando $\omega_0^2 = k/m$ se obtiene~\eqref{eq:oas}.
\end{proof}

\begin{proposicion}[Solución general del OAS]
La solución general de~\eqref{eq:oas} es:
\begin{mdframed}[style=resultado]
\begin{equation}
  x(t) = A\,\sen(\omega_0 t - \delta)
  \label{eq:sol-oas}
\end{equation}
\end{mdframed}
donde $A \geq 0\,[\mathrm{m}]$ es la \emph{amplitud} y $\delta\,[\mathrm{rad}]$ es la
\emph{fase inicial}. Ambas constantes se determinan con las condiciones iniciales
$x(0)$ y $\dot{x}(0)$.
\end{proposicion}

\begin{proof}
La ecuación~\eqref{eq:oas} es una EDO lineal de segundo orden con coeficientes constantes.
Proponiendo $x = e^{rt}$ se obtiene la ecuación característica $r^2 + \omega_0^2 = 0$, de
donde $r = \pm i\omega_0$. La solución real general es:
\[
  x(t) = C_1\cos\omega_0 t + C_2\sen\omega_0 t
       = A\,\sen(\omega_0 t - \delta),
\]
con $A = \sqrt{C_1^2 + C_2^2}$ y $\tan\delta = -C_1/C_2$.
\end{proof}

El movimiento es periódico con \emph{período} $T = 2\pi/\omega_0$ y
\emph{frecuencia} $\nu_0 = \omega_0/(2\pi)\,[\mathrm{Hz}]$.

\subsubsection{Energía del oscilador armónico}

La energía mecánica total se conserva durante el movimiento:

\begin{proposicion}[Conservación de la energía]
\begin{mdframed}[style=resultado]
\begin{equation}
  E = \frac{1}{2}m\dot{x}^2 + \frac{1}{2}kx^2 = \frac{1}{2}kA^2 = \frac{1}{2}m\omega_0^2 A^2
  = \mathrm{cte}
  \label{eq:energia-oas}
\end{equation}
\end{mdframed}
\end{proposicion}

\begin{proof}
Sustituyendo $x(t) = A\sen(\omega_0 t - \delta)$ y $\dot{x}(t) = A\omega_0\cos(\omega_0 t - \delta)$:
\[
  E = \tfrac{1}{2}mA^2\omega_0^2\cos^2(\omega_0 t - \delta)
      + \tfrac{1}{2}kA^2\sen^2(\omega_0 t - \delta)
    = \tfrac{1}{2}kA^2\bigl[\cos^2(\cdot) + \sen^2(\cdot)\bigr]
    = \tfrac{1}{2}kA^2,
\]
donde se usó $m\omega_0^2 = k$.
\end{proof}

La energía cinética y potencial oscilan con frecuencia $2\omega_0$, pero su suma permanece
constante. La energía es proporcional al cuadrado de la amplitud.

\begin{nota}
En la realidad, las fuerzas disipativas acaban extinguiendo el movimiento. El OAS libre es
una idealización que resulta válida en ausencia de disipación.
\end{nota}

\subsection{Oscilaciones amortiguadas}

En presencia de una fuerza de rozamiento viscoso $F_{\mathrm{r}} = -b\dot{x}$
(proporcional a la velocidad, con $b > 0\,[\mathrm{kg\,s^{-1}}]$), la ecuación de
movimiento es:
\[
  m\ddot{x} = -kx - b\dot{x}
\]

\begin{definicion}[Coeficiente de amortiguamiento]
Se define el \emph{coeficiente de amortiguamiento}:
\begin{equation}
  \beta \equiv \frac{b}{2m} \quad [\mathrm{s^{-1}}]
  \label{eq:beta}
\end{equation}
de modo que la ecuación del oscilador amortiguado queda:
\begin{mdframed}[style=resultado]
\begin{equation}
  \ddot{x} + 2\beta\dot{x} + \omega_0^2\,x = 0
  \label{eq:amortiguada}
\end{equation}
\end{mdframed}
\end{definicion}

\begin{proof}[Solución general de la ecuación amortiguada]
Proponiendo $x(t) = e^{rt}$ se obtiene la \emph{ecuación característica}:
\begin{equation}
  r^2 + 2\beta r + \omega_0^2 = 0 \implies r = -\beta \pm \sqrt{\beta^2 - \omega_0^2}
  \label{eq:car-amor}
\end{equation}
El carácter de las raíces depende del signo de $\beta^2 - \omega_0^2$.
\end{proof}

Se distinguen tres regímenes según la relación entre $\omega_0$ y $\beta$:

\subsubsection{Régimen subamortiguado \texorpdfstring{($\omega_0^2 > \beta^2$)}{}}

Las raíces son complejas conjugadas. Definiendo la \emph{frecuencia amortiguada}:
\begin{equation}
  \Omega \equiv \sqrt{\omega_0^2 - \beta^2} < \omega_0 \quad [\mathrm{rad\,s^{-1}}]
  \label{eq:omega-amor}
\end{equation}
la solución general es:
\begin{mdframed}[style=resultado]
\begin{equation}
  x(t) = A\,e^{-\beta t}\cos(\Omega t + \varphi)
  \label{eq:sol-subamor}
\end{equation}
\end{mdframed}
donde $A$ y $\varphi$ se fijan con las condiciones iniciales. La amplitud decae como
$e^{-\beta t}$ mientras el sistema oscila con frecuencia reducida $\Omega$.

\begin{definicion}[Factor de calidad]
El \emph{factor de calidad} $Q$ mide la eficiencia energética del oscilador (número de
ciclos que tarda en amortiguarse significativamente):
\begin{equation}
  Q \equiv \frac{\omega_0}{2\beta}
  \label{eq:Q}
\end{equation}
Un sistema con $Q \gg 1$ está poco amortiguado; con $Q \approx 1/2$ se acerca al
amortiguamiento crítico.
\end{definicion}

\subsubsection{Amortiguamiento crítico \texorpdfstring{($\omega_0^2 = \beta^2$)}{}}

Las dos raíces coinciden: $r_{1,2} = -\beta$ (raíz doble). La solución general es:
\begin{mdframed}[style=resultado]
\begin{equation}
  x(t) = (A + Bt)\,e^{-\beta t}
  \label{eq:sol-critico}
\end{equation}
\end{mdframed}
donde $A$ y $B$ se determinan con las condiciones iniciales. El sistema retorna al
equilibrio en el menor tiempo posible sin oscilar: es el régimen óptimo para absorción de
vibraciones (amortiguadores de automóvil, puertas hidráulicas).

\subsubsection{Régimen sobreamortiguado \texorpdfstring{($\omega_0^2 < \beta^2$)}{}}

Las dos raíces son reales y negativas. Definiendo:
\begin{equation}
  \gamma \equiv \sqrt{\beta^2 - \omega_0^2} < \beta \quad [\mathrm{rad\,s^{-1}}]
  \label{eq:gamma}
\end{equation}
la solución general es:
\begin{mdframed}[style=resultado]
\begin{equation}
  x(t) = e^{-\beta t}\!\left[A\,e^{\gamma t} + B\,e^{-\gamma t}\right]
  \label{eq:sol-sobreamo}
\end{equation}
\end{mdframed}
El sistema vuelve al equilibrio sin oscilar, pero más lentamente que en el caso crítico.

\begin{observacion}
Entre los tres regímenes, el amortiguamiento crítico proporciona la vuelta al equilibrio más
rápida. Para $\beta$ fijo, el sobreamortiguado es más lento porque el exponente dominante
es $e^{-(\beta-\gamma)t}$, con $\beta - \gamma < \beta$.
\end{observacion}

\subsection{Oscilaciones forzadas}

Cuando actúa sobre el oscilador amortiguado una fuerza periódica externa
$F(t) = F_0\cos\omega t$ ($F_0\,[\mathrm{N}]$, $\omega\,[\mathrm{rad\,s^{-1}}]$), la
ecuación de movimiento es:

\begin{mdframed}[style=resultado]
\begin{equation}
  \ddot{x} + 2\beta\dot{x} + \omega_0^2\,x = \frac{F_0}{m}\cos\omega t
  \label{eq:forzada}
\end{equation}
\end{mdframed}

La solución general es la suma de la solución homogénea (transitorio, que decae en un
tiempo $\sim 1/\beta$) y una solución particular periódica (régimen estacionario):
\begin{equation}
  x(t) = x_{\mathrm{trans}}(t) + x_p(t), \qquad x_p(t) = D\cos(\omega t - \delta)
  \label{eq:sol-forzada}
\end{equation}

\begin{proposicion}[Amplitud y desfase en régimen estacionario]
La amplitud estacionaria $D$ y el desfase $\delta$ valen:
\begin{mdframed}[style=resultado]
\begin{align}
  D &= \frac{F_0/m}{\sqrt{(\omega_0^2-\omega^2)^2 + 4\beta^2\omega^2}}
  \label{eq:amp-forzada} \\[6pt]
  \tan\delta &= \frac{2\beta\omega}{\omega_0^2 - \omega^2}
  \label{eq:fase-forzada}
\end{align}
\end{mdframed}
\end{proposicion}

\begin{proof}
Se sustituye $x_p = D\cos(\omega t - \delta)$ en~\eqref{eq:forzada}. Calculando
$\dot{x}_p = -D\omega\sen(\omega t - \delta)$ y
$\ddot{x}_p = -D\omega^2\cos(\omega t - \delta)$:
\begin{align}
  -D\omega^2\cos(\omega t-\delta)
  - 2\beta D\omega\sen(\omega t-\delta)
  + \omega_0^2 D\cos(\omega t-\delta)
  &= \frac{F_0}{m}\cos\omega t
\end{align}
Expandiendo $\cos(\omega t - \delta) = \cos\omega t\cos\delta + \sen\omega t\sen\delta$ y
$\sen(\omega t - \delta) = \sen\omega t\cos\delta - \cos\omega t\sen\delta$, e igualando
los coeficientes de $\cos\omega t$ y $\sen\omega t$:
\begin{align}
  \cos\omega t:& \quad D\bigl[(\omega_0^2-\omega^2)\cos\delta + 2\beta\omega\sen\delta\bigr]
                      = \frac{F_0}{m} \\
  \sen\omega t:& \quad D\bigl[(\omega_0^2-\omega^2)\sen\delta - 2\beta\omega\cos\delta\bigr]
                      = 0
\end{align}
De la segunda ecuación: $\tan\delta = 2\beta\omega/(\omega_0^2-\omega^2)$.
Sustituyendo en la primera y usando $1 + \tan^2\delta = 1/\cos^2\delta$ se obtiene~\eqref{eq:amp-forzada}.
\end{proof}

\begin{definicion}[Resonancia]
Se produce \emph{resonancia} cuando la frecuencia de la fuerza $\omega$ se acerca a la
frecuencia natural $\omega_0$. Para $\beta \ll \omega_0$, la amplitud estacionaria $D$
alcanza su máximo (que diverge para $\beta \to 0$) en la \emph{frecuencia de resonancia}:
\begin{equation}
  \omega_{\mathrm{res}} = \sqrt{\omega_0^2 - 2\beta^2}
  \label{eq:omega-res}
\end{equation}
\end{definicion}

\subsection{Ejemplos}

\begin{ejemplo}[Oscilador amortiguado]
Una masa $m = 1\,\mathrm{kg}$ está unida a un resorte $k = 25\,\mathrm{N\,m^{-1}}$ sobre
una superficie horizontal con coeficiente $b = 6\,\mathrm{kg\,s^{-1}}$. Se suelta desde
el reposo a $x(0) = 0{,}10\,\mathrm{m}$.

\medskip
De~\eqref{eq:beta}: $\beta = b/(2m) = 3\,\mathrm{s^{-1}}$; $\omega_0^2 = 25\,\mathrm{rad^2\,s^{-2}}$.
Como $\omega_0^2 = 25 > \beta^2 = 9$, el sistema es \textbf{subamortiguado} y de~\eqref{eq:omega-amor}:
\[
  \Omega = \sqrt{25 - 9} = 4\,\mathrm{rad\,s^{-1}}
\]
La solución general~\eqref{eq:sol-subamor} escrita en forma de suma es
$x(t) = e^{-3t}(C_1\cos 4t + C_2\sen 4t)$.
Aplicando $x(0) = 0{,}10$ y $\dot{x}(0) = 0$:
\[
  C_1 = 0{,}10\,\mathrm{m}, \qquad C_2 = \frac{\beta C_1}{\Omega} = 0{,}075\,\mathrm{m}
\]
\[
  \boxed{x(t) = e^{-3t}\!\left(0{,}10\cos 4t + 0{,}075\sen 4t\right)\,\mathrm{m}}
\]
\end{ejemplo}

\begin{ejemplo}[Oscilador forzado]
Mismos datos pero $b = 4\,\mathrm{kg\,s^{-1}}$ y fuerza $F(t) = 10\cos(3t)\,\mathrm{N}$
aplicada desde $t = 0$; la masa parte del reposo en $x(0) = 0{,}10\,\mathrm{m}$.

\medskip
$\beta = 2\,\mathrm{s^{-1}}$, $\omega_0^2 = 25$, $\omega = 3\,\mathrm{rad\,s^{-1}}$,
$F_0/m = 10\,\mathrm{m\,s^{-2}}$.

Amplitud estacionaria~\eqref{eq:amp-forzada}:
\[
  D = \frac{10}{\sqrt{(25-9)^2 + 4 \cdot 4 \cdot 9}}
    = \frac{10}{\sqrt{256+144}} = \frac{10}{20} = 0{,}5\,\mathrm{m}
\]
Desfase~\eqref{eq:fase-forzada}: $\tan\delta = \frac{2\cdot2\cdot3}{25-9} = \frac{3}{4}
\Rightarrow \delta \approx 0{,}6435\,\mathrm{rad}$.

La solución completa es $x(t) = x_{\mathrm{trans}}(t) + 0{,}5\cos(3t - 0{,}6435)$, donde
el transitorio $x_{\mathrm{trans}}$ decae con $e^{-2t}$ y se determina aplicando las
condiciones iniciales a la solución total.
\end{ejemplo}

\section{Tema 1 (continuación): Osciladores Acoplados}
% ══════════════════════════════════════════════════════════════════════════

\subsection{Introducción}

\begin{definicion}[Osciladores acoplados]
Un sistema de \emph{osciladores acoplados} está formado por dos o más osciladores que
interactúan entre sí mediante algún mecanismo de acoplamiento (fuerza mecánica, campo
eléctrico, interacción biológica, etc.). El comportamiento de cada oscilador depende
entonces tanto de sus propiedades individuales como de la influencia de los demás.
\end{definicion}

Un sistema físico real puede vibrar en muchos modos distintos y resonar a múltiples
frecuencias. Los tres tipos de oscilador base son:
\begin{itemize}
  \item \textbf{Oscilador armónico simple:} $x(t) = A\,\sen(\omega t + \delta)$
  \item \textbf{Oscilador amortiguado:}
        $x(t) = A_0\,e^{-\beta t}\sen(\Omega t + \delta)$
  \item \textbf{Oscilador forzado:}
        $x(t) = A_0\,e^{-\beta t}\sen(\Omega t + \delta) + D\cos(\omega t - \delta')$
\end{itemize}

\subsection{Dos osciladores acoplados: ecuaciones de movimiento}

Consideremos dos péndulos idénticos de longitud $l$ y masa $m$, unidos por un resorte
de constante $k$ conectado en sus posiciones de equilibrio. Sean $x_A$ y $x_B$ los
desplazamientos horizontales de cada masa respecto al equilibrio.

Las fuerzas que actúan sobre cada masa son:
\begin{itemize}
  \item La fuerza restauradora del péndulo: $-m\omega_0^2 x$, con
        $\omega_0 \equiv \sqrt{g/l}\,[\mathrm{rad\,s^{-1}}]$.
  \item La fuerza del resorte de acoplamiento: $\pm k(x_A - x_B)$.
\end{itemize}

Definiendo la \emph{frecuencia de acoplamiento}:
\begin{equation}
  \omega_c \equiv \sqrt{\frac{k}{m}} \quad [\mathrm{rad\,s^{-1}}]
  \label{eq:omega-c}
\end{equation}
y aplicando la segunda ley de Newton a cada masa, las ecuaciones de movimiento son:

\begin{mdframed}[style=resultado]
\begin{align}
  \ddot{x}_A + (\omega_0^2 + \omega_c^2)\,x_A - \omega_c^2\,x_B &= 0
  \label{eq:acop-A} \\[4pt]
  \ddot{x}_B + (\omega_0^2 + \omega_c^2)\,x_B - \omega_c^2\,x_A &= 0
  \label{eq:acop-B}
\end{align}
\end{mdframed}

Estas dos ecuaciones están acopladas y no pueden resolverse independientemente.

\subsection{Modos normales: consideraciones de simetría}

\begin{definicion}[Modo normal de vibración]
Un \emph{modo normal} es una solución en la que todas las partículas del sistema oscilan
sincrónicamente a la \emph{misma} frecuencia y con relación de amplitudes constante en el
tiempo.
\end{definicion}

La simetría del sistema (dos péndulos idénticos) permite encontrar los modos por inspección.

\subsubsection{Modo simétrico (en fase): \texorpdfstring{$\omega_1 = \omega_0$}{ω₁=ω₀}}

Si $x_A = x_B$ el resorte de acoplamiento no se deforma y su contribución es nula.
Cada péndulo oscila libremente:
\begin{equation}
  x_A(t) = x_B(t) = C\cos\omega_0 t, \qquad \omega_1 = \omega_0 = \sqrt{g/l}
  \label{eq:modo-sim}
\end{equation}

\subsubsection{Modo antisimétrico (en antifase): \texorpdfstring{$\omega_2 = \sqrt{\omega_0^2 + 2\omega_c^2}$}{}}

Si $x_A = -x_B$ el resorte se estira/comprime al máximo y proporciona una fuerza
restauradora adicional:
\begin{equation}
  x_A(t) = C\cos\omega_2 t,\quad x_B(t) = -C\cos\omega_2 t,
  \qquad \omega_2 = \sqrt{\omega_0^2 + 2\omega_c^2}
  \label{eq:modo-antis}
\end{equation}
Siempre $\omega_2 > \omega_1$.

\subsection{Coordenadas normales}

\begin{definicion}[Coordenadas normales]
Las \emph{coordenadas normales} son las combinaciones lineales de las coordenadas físicas
que desacoplan el sistema. Para los dos péndulos acoplados:
\begin{mdframed}[style=resultado]
\begin{equation}
  q_1 \equiv x_A + x_B, \qquad q_2 \equiv x_A - x_B
  \label{eq:coord-norm}
\end{equation}
\end{mdframed}
\end{definicion}

\begin{proposicion}[Desacoplamiento en coordenadas normales]
En las coordenadas $q_1$, $q_2$ el sistema~\eqref{eq:acop-A}--\eqref{eq:acop-B} se
reduce a dos osciladores independientes:
\begin{mdframed}[style=resultado]
\begin{align}
  \ddot{q}_1 + \omega_1^2\,q_1 &= 0, \qquad \omega_1^2 = \omega_0^2
  \label{eq:q1} \\[4pt]
  \ddot{q}_2 + \omega_2^2\,q_2 &= 0, \qquad \omega_2^2 = \omega_0^2 + 2\omega_c^2
  \label{eq:q2}
\end{align}
\end{mdframed}
\end{proposicion}

\begin{proof}
Sumando~\eqref{eq:acop-A} y~\eqref{eq:acop-B}:
\[
  (\ddot{x}_A + \ddot{x}_B) + (\omega_0^2+\omega_c^2)(x_A+x_B)
  - \omega_c^2(x_B+x_A) = 0
  \implies \ddot{q}_1 + \omega_0^2 q_1 = 0.
\]
Restando~\eqref{eq:acop-B} de~\eqref{eq:acop-A}:
\[
  (\ddot{x}_A - \ddot{x}_B) + (\omega_0^2+\omega_c^2)(x_A-x_B)
  + \omega_c^2(x_A-x_B) = 0
  \implies \ddot{q}_2 + (\omega_0^2+2\omega_c^2)q_2 = 0.
\]
\end{proof}

Las soluciones de~\eqref{eq:q1}--\eqref{eq:q2} son $q_1 = C\cos\omega_1 t$ y
$q_2 = D\cos\omega_2 t$. Invirtiendo~\eqref{eq:coord-norm}:
\begin{mdframed}[style=resultado]
\begin{align}
  x_A(t) &= \frac{1}{2}(q_1 + q_2) = \frac{C}{2}\cos\omega_1 t + \frac{D}{2}\cos\omega_2 t
  \label{eq:xA} \\[4pt]
  x_B(t) &= \frac{1}{2}(q_1 - q_2) = \frac{C}{2}\cos\omega_1 t - \frac{D}{2}\cos\omega_2 t
  \label{eq:xB}
\end{align}
\end{mdframed}

\begin{itemize}
  \item $C = 0$: solo el modo antisimétrico ($\omega_2$).
  \item $D = 0$: solo el modo simétrico ($\omega_1$).
\end{itemize}

\subsubsection{Batidos}

Cuando $x_A(0) = A_0$, $x_B(0) = 0$, $\dot{x}_A(0) = \dot{x}_B(0) = 0$, las condiciones
iniciales dan $C = D = A_0$. Usando las identidades de producto a suma:

\begin{mdframed}[style=resultado]
\begin{align}
  x_A(t) &= A_0\cos\!\left(\frac{\omega_2-\omega_1}{2}\,t\right)
              \cos\!\left(\frac{\omega_2+\omega_1}{2}\,t\right)
  \label{eq:batidos-A} \\[4pt]
  x_B(t) &= A_0\sen\!\left(\frac{\omega_2-\omega_1}{2}\,t\right)
              \sen\!\left(\frac{\omega_2+\omega_1}{2}\,t\right)
  \label{eq:batidos-B}
\end{align}
\end{mdframed}

Para acoplamiento débil ($\omega_c \ll \omega_0$) la diferencia de frecuencias
$\omega_2 - \omega_1 \ll \omega_1$: la amplitud de $x_A$ oscila lentamente con frecuencia
de modulación $(\omega_2-\omega_1)/2$ y la energía se transfiere periódicamente de A a B
y viceversa (\emph{batidos}).

\subsection{Método analítico general}

Para resolver el sistema sin recurrir a la simetría se propone la \emph{solución de modo normal}:
\begin{equation}
  x_A = C\cos\omega t, \qquad x_B = C'\cos\omega t
  \label{eq:propuesta}
\end{equation}

Sustituyendo en~\eqref{eq:acop-A}--\eqref{eq:acop-B}:
\begin{align}
  \bigl(-\omega^2 + \omega_0^2 + \omega_c^2\bigr)C - \omega_c^2 C' &= 0
  \label{eq:ana-A} \\[4pt]
  -\omega_c^2 C + \bigl(-\omega^2 + \omega_0^2 + \omega_c^2\bigr)C' &= 0
  \label{eq:ana-B}
\end{align}

\begin{proposicion}[Condición de modos normales]
Para que~\eqref{eq:ana-A}--\eqref{eq:ana-B} tengan solución no trivial, el determinante
de la matriz de coeficientes debe ser cero:
\begin{equation}
  \begin{vmatrix}
    -\omega^2+\omega_0^2+\omega_c^2 & -\omega_c^2 \\
    -\omega_c^2 & -\omega^2+\omega_0^2+\omega_c^2
  \end{vmatrix} = 0
  \label{eq:det}
\end{equation}
\end{proposicion}

\begin{proof}
La condición~\eqref{eq:det} da:
\[
  \bigl(-\omega^2+\omega_0^2+\omega_c^2\bigr)^2 - \omega_c^4 = 0
  \implies
  -\omega^2+\omega_0^2+\omega_c^2 = \pm\omega_c^2
\]
Los dos casos son:
\begin{itemize}
  \item $+\omega_c^2$: $\omega^2 = \omega_0^2 \implies \omega_1 = \omega_0$, con
        $C/C' = +1$ (modo simétrico).
  \item $-\omega_c^2$: $\omega^2 = \omega_0^2 + 2\omega_c^2 \implies \omega_2$, con
        $C/C' = -1$ (modo antisimétrico).
\end{itemize}
\end{proof}

La solución general resulta entonces:
\begin{equation}
  x_A = C\cos\omega_1 t + D\cos\omega_2 t, \qquad
  x_B = C\cos\omega_1 t - D\cos\omega_2 t
  \label{eq:sol-gen-acop}
\end{equation}

\subsection{Oscilaciones forzadas en sistemas acoplados}

Supongamos que sobre la masa A actúa $F(t) = F_0\cos\omega t$. Las ecuaciones son:
\begin{align}
  \ddot{x}_A + (\omega_0^2+\omega_c^2)x_A - \omega_c^2 x_B &= \frac{F_0}{m}\cos\omega t
  \label{eq:forzado-A} \\[4pt]
  \ddot{x}_B + (\omega_0^2+\omega_c^2)x_B - \omega_c^2 x_A &= 0
  \label{eq:forzado-B}
\end{align}

En coordenadas normales~\eqref{eq:coord-norm}, sumando y restando:
\begin{align}
  \ddot{q}_1 + \omega_1^2\,q_1 &= \frac{F_0}{m}\cos\omega t
  \label{eq:forzado-q1} \\[4pt]
  \ddot{q}_2 + \omega_2^2\,q_2 &= \frac{F_0}{m}\cos\omega t
  \label{eq:forzado-q2}
\end{align}

Las soluciones estacionarias de cada oscilador desacoplado son:
\begin{equation}
  q_1 = \frac{F_0/m}{\omega_1^2 - \omega^2}\cos\omega t \equiv C\cos\omega t,\qquad
  q_2 = \frac{F_0/m}{\omega_2^2 - \omega^2}\cos\omega t \equiv D\cos\omega t
  \label{eq:q-forzado}
\end{equation}

Las amplitudes de desplazamiento resultan:
\begin{mdframed}[style=resultado]
\begin{equation}
  x_A = \frac{C+D}{2}\cos\omega t, \qquad
  x_B = \frac{C-D}{2}\cos\omega t
  \label{eq:amp-acop}
\end{equation}
\end{mdframed}

Se producen \textbf{resonancias} en $\omega = \omega_1$ (diverge $C$) y
en $\omega = \omega_2$ (diverge $D$): el sistema tiene exactamente dos frecuencias de
resonancia, correspondientes a sus dos modos normales.

\begin{ejemplo}[Tres resortes, dos masas]
Sistema pared$-k-2m-k-2m-k-$pared con tres resortes idénticos de constante $k$ y dos
masas iguales $2m$.

Las ecuaciones de movimiento son $2m\ddot{x}_A + 2kx_A - kx_B = 0$ y análoga para B.
Con $\omega_0^2 = k/(2m)$ y $\omega_c^2 = k/(2m)$ se aplica~\eqref{eq:det}:
\[
  \omega_1 = \sqrt{\frac{k}{2m}} \quad (\text{modo simétrico}),\qquad
  \omega_2 = \sqrt{\frac{3k}{2m}} \quad (\text{modo antisimétrico})
\]
\end{ejemplo}

\begin{ejemplo}[Dos péndulos con muelle colgante]
Una partícula de masa $m$ cuelga de un resorte $\kappa$; a esta masa se engancha otro
sistema igual. Las ecuaciones de movimiento son idénticas a las de los péndulos acoplados
con $\omega_0^2 = \kappa/m$ y $\omega_c^2 = \kappa/m$ (el resorte central es a la vez
restaurador y acoplador). Por tanto:
\[
  \omega_1 = \omega_0 = \sqrt{\kappa/m},\qquad \omega_2 = \sqrt{3\kappa/m}
\]
Si una masa se fija, la otra oscila con $\omega_{\text{fija}} = \sqrt{2\kappa/m}$, que
queda entre $\omega_1$ y $\omega_2$.
\end{ejemplo}

\subsection{\texorpdfstring{$N$}{N} osciladores acoplados: la cuerda vibrante discreta}

\subsubsection{Principio de superposición de Bernoulli}

Daniel Bernoulli demostró en 1753 que un sistema de $N$ masas idénticas tiene exactamente
$N$ modos de vibración independientes, y enunció el
\textbf{principio de superposición}: \emph{el movimiento general es la superposición
lineal de todos los modos normales}.

\subsubsection{Ecuación de movimiento}

\begin{definicion}[Cuerda vibrante discreta]
Consideremos $N$ partículas de masa $m$ equiespaciadas una distancia $l$ a lo largo de una
cuerda sin masa de tensión $T$, con los extremos $p=0$ y $p=N+1$ fijos. Sea $y_p$ el
desplazamiento transversal de la partícula $p$.
\end{definicion}

Para ángulos pequeños ($\sen\alpha \approx \tan\alpha$), la fuerza transversal neta sobre
la partícula $p$ es:
\[
  F_p = T\!\left(\frac{y_{p+1}-y_p}{l}\right) - T\!\left(\frac{y_p - y_{p-1}}{l}\right)
      = \frac{T}{l}\,(y_{p+1} - 2y_p + y_{p-1})
\]

\begin{proposicion}[Ecuación de movimiento de la cuerda discreta]
\begin{mdframed}[style=resultado]
\begin{equation}
  \ddot{y}_p + 2\omega_0^2\,y_p - \omega_0^2(y_{p+1}+y_{p-1}) = 0,
  \qquad \omega_0^2 \equiv \frac{T}{ml},\quad p = 1,\ldots,N
  \label{eq:cuerda-discreta}
\end{equation}
\end{mdframed}
con condiciones de contorno $y_0 = y_{N+1} = 0$.
\end{proposicion}

\subsubsection{Cálculo de los \texorpdfstring{$N$}{N} modos normales}

Se propone $y_p(t) = A_p\cos\omega t$. Sustituyendo en~\eqref{eq:cuerda-discreta}:
\begin{equation}
  (-\omega^2 + 2\omega_0^2)A_p = \omega_0^2(A_{p-1}+A_{p+1})
  \implies \frac{A_{p-1}+A_{p+1}}{A_p} = \frac{-\omega^2+2\omega_0^2}{\omega_0^2}
  \label{eq:recurrencia}
\end{equation}

Se propone la función espacial $A_p = C\sen(p\theta)$, que satisface la recurrencia con
$2\cos\theta = (-\omega^2+2\omega_0^2)/\omega_0^2$. La condición $y_0 = 0$ se cumple
automáticamente; la condición $y_{N+1} = 0$ exige:
\[
  \sen\!\bigl((N+1)\theta\bigr) = 0
  \implies \theta_n = \frac{n\pi}{N+1}, \quad n = 1, 2, \ldots, N
\]

\begin{proposicion}[Formas de los modos normales]
El perfil de amplitud del modo $n$ es:
\begin{mdframed}[style=resultado]
\begin{equation}
  A_p^{(n)} = C_n\,\sen\!\left(\frac{n\,p\,\pi}{N+1}\right), \qquad p = 1,\ldots,N
  \label{eq:perfil}
\end{equation}
\end{mdframed}
\end{proposicion}

\begin{proposicion}[Frecuencias de los modos normales]
Las $N$ frecuencias de los modos normales son:
\begin{mdframed}[style=resultado]
\begin{equation}
  \omega_n = 2\omega_0\,\sen\!\left(\frac{n\pi}{2(N+1)}\right), \qquad n = 1,2,\ldots,N
  \label{eq:frec-discreta}
\end{equation}
\end{mdframed}
\end{proposicion}

\begin{proof}
De la relación de dispersión $2\cos\theta_n = (2\omega_0^2 - \omega_n^2)/\omega_0^2$:
\[
  \omega_n^2 = 2\omega_0^2(1 - \cos\theta_n) = 4\omega_0^2\sen^2\!\left(\frac{\theta_n}{2}\right)
  = 4\omega_0^2\sen^2\!\left(\frac{n\pi}{2(N+1)}\right)
\]
tomando raíz positiva se obtiene~\eqref{eq:frec-discreta}.
\end{proof}

El movimiento de la partícula $p$ en el modo $n$ es:
\begin{equation}
  y_{p}^{(n)}(t) = C_n\,\sen\!\left(\frac{n\,p\,\pi}{N+1}\right)\cos\omega_n t
  \label{eq:modo-n}
\end{equation}

El modo $n$ tiene $n-1$ \emph{nodos internos} (puntos fijos distintos de los extremos).
Los modos con $n > N$ repiten los patrones de los modos $n \leq N$ con signo opuesto.

\begin{nota}
El mismo formalismo describe oscilaciones longitudinales: basta sustituir $y_p$ por el
desplazamiento longitudinal $\xi_p$. Las ecuaciones y frecuencias son idénticas.
\end{nota}

\subsection{Límite continuo: la cuerda vibrante continua}

\subsubsection{Del sistema discreto al continuo}

Cuando $N \to \infty$ manteniendo fija la longitud $L = (N+1)l$ y la densidad lineal
$\mu = m/l\,[\mathrm{kg\,m^{-1}}]$, el sistema discreto converge a una cuerda continua.
Para modos de baja frecuencia ($n \ll N$):
\[
  \frac{n\pi}{2(N+1)} \ll 1 \implies \sen\!\left(\frac{n\pi}{2(N+1)}\right)
  \approx \frac{n\pi}{2(N+1)}
\]
y de~\eqref{eq:frec-discreta}:
\begin{equation}
  \omega_n \xrightarrow{N\to\infty} \frac{n\pi}{L}\sqrt{\frac{T}{\mu}} = n\omega_1
  \label{eq:limite-cont}
\end{equation}
con $\omega_1 = \pi v/L$ y $v = \sqrt{T/\mu}$ la velocidad de propagación.

Los desplazamientos pasan a función continua $y_p \to y(x,t)$ con $x = pl$:
\begin{mdframed}[style=resultado]
\begin{equation}
  y_n(x,t) = C_n\,\sen\!\left(\frac{n\pi x}{L}\right)\cos\omega_n t
  \label{eq:modo-cont}
\end{equation}
\end{mdframed}

\subsubsection{Derivación de la ecuación de ondas}

\begin{proposicion}[Ecuación de ondas en la cuerda continua]
El desplazamiento transversal $y(x,t)$ de una cuerda de densidad lineal $\mu$ y tensión
$T$ satisface:
\begin{mdframed}[style=resultado]
\begin{equation}
  \frac{\partial^2 y}{\partial x^2} = \frac{1}{v^2}\frac{\partial^2 y}{\partial t^2},
  \qquad v \equiv \sqrt{\frac{T}{\mu}} \quad [\mathrm{m\,s^{-1}}]
  \label{eq:onda}
\end{equation}
\end{mdframed}
\end{proposicion}

\begin{proof}
Consideremos un elemento de cuerda de longitud $\Delta x$ y masa $\mu\,\Delta x$.
Los extremos forman ángulos $\theta$ y $\theta + \Delta\theta$ con la horizontal. Para
ángulos pequeños la fuerza horizontal es nula y la fuerza transversal neta es:
\[
  F_y = T\sen(\theta+\Delta\theta) - T\sen\theta
      \approx T\,\Delta\theta
      = T\,\frac{\partial^2 y}{\partial x^2}\,\Delta x
\]
donde se usó $\tan\theta = \partial y/\partial x$ y para ángulos pequeños
$\Delta\theta \approx (\partial^2 y/\partial x^2)\,\Delta x$.
Aplicando la segunda ley de Newton al elemento:
\[
  \mu\,\Delta x\,\frac{\partial^2 y}{\partial t^2}
  = T\,\frac{\partial^2 y}{\partial x^2}\,\Delta x
  \implies
  \frac{\partial^2 y}{\partial x^2} = \frac{\mu}{T}\frac{\partial^2 y}{\partial t^2}
  = \frac{1}{v^2}\frac{\partial^2 y}{\partial t^2}
\]
\end{proof}

\subsubsection{Modos normales de la cuerda continua}

Buscamos soluciones de variables separadas $y(x,t) = f(x)\cos\omega t$. Sustituyendo
en~\eqref{eq:onda}:
\begin{equation}
  \frac{d^2f}{dx^2} + \frac{\omega^2}{v^2}\,f = 0
  \label{eq:helmholtz}
\end{equation}
La solución es $f(x) = A\sen(\omega x/v + \phi)$. Las condiciones de contorno
$y(0,t) = y(L,t) = 0$ imponen $\phi = 0$ y:
\begin{equation}
  \frac{\omega L}{v} = n\pi \implies \omega_n = n\frac{\pi v}{L} = n\omega_1,
  \quad n = 1, 2, 3, \ldots
  \label{eq:frec-cont}
\end{equation}

\begin{proposicion}[Frecuencias y longitudes de onda de los modos normales]
\begin{mdframed}[style=resultado]
\begin{equation}
  \nu_n = \frac{nv}{2L} = \frac{n}{2L}\sqrt{\frac{T}{\mu}},
  \qquad
  \lambda_n = \frac{2L}{n}
  \label{eq:nu-lambda}
\end{equation}
\end{mdframed}
donde $n = 1$ es el \emph{fundamental} (o primer armónico) y $n \geq 2$ son los
\emph{armónicos} o \emph{sobretonos}.
\end{proposicion}

Los nodos del modo $n$ están en $x = kL/n$ ($k = 0, 1, \ldots, n$); los antinodos en
los puntos intermedios. Cualquier movimiento de la cuerda puede expresarse como
superposición de estos modos (principio de Bernoulli).

\subsubsection{Cuerda forzada en un extremo}

Con las condiciones de contorno $y(0,t) = B\cos\omega t$ y $y(L,t) = 0$, la solución
particular en régimen estacionario es:
\begin{equation}
  y(x,t) = A\,\sen\!\left(\frac{\omega x}{v} + \alpha\right)\cos\omega t
  \label{eq:forzada-cuerda}
\end{equation}
Imponiendo $y(L,t) = 0$ y $y(0,t) = B\cos\omega t$:
\[
  A = \frac{B}{\sen\!\left(p\pi - \dfrac{\omega L}{v}\right)}, \quad
  \alpha = p\pi - \frac{\omega L}{v}
\]
La amplitud diverge cuando $\omega L/v = p\pi$, es decir, $\omega = \omega_p$:
\textbf{resonancia} con el modo $p$.

\begin{ejemplo}[Cuerda con \texorpdfstring{$N=2$}{N=2} masas]
Cuerda de longitud $3l$ con dos masas $m$ en $x = l$ y $x = 2l$, con $\omega_0 = \sqrt{T/(ml)}$.
De~\eqref{eq:frec-discreta} con $N = 2$:
\[
  \omega_1 = 2\omega_0\sen\!\left(\frac{\pi}{6}\right) = \omega_0
  \quad\text{(modo inferior, en fase)}
\]
\[
  \omega_2 = 2\omega_0\sen\!\left(\frac{2\pi}{6}\right) = \omega_0\sqrt{3}
  \quad\text{(modo superior, en antifase)}
\]
Para $N = 1$ (una masa en $x = l$, longitud $2l$):
$\omega_1 = 2\omega_0\sen(\pi/4) = \omega_0\sqrt{2}$.
\end{ejemplo}

\begin{ejemplo}[Osciladores acoplados forzados: pared móvil]
Sistema de dos masas $m$ unidas entre sí y a las paredes por resortes de constante $k$.
La pared izquierda oscila según $\eta(t) = \eta_0\cos\omega t$.

Las ecuaciones de movimiento son:
\begin{align}
  m\ddot{x}_1 &= k\eta - 2kx_1 + kx_2 \\
  m\ddot{x}_2 &= k(x_1 - x_2)
\end{align}

En coordenadas normales $q_1 = x_1+x_2$, $q_2 = x_1-x_2$ con $\omega_0^2 = k/m$:
\begin{align}
  \ddot{q}_1 + \omega_0^2 q_1 &= \omega_0^2\eta_0\cos\omega t \\
  \ddot{q}_2 + 3\omega_0^2 q_2 &= \omega_0^2\eta_0\cos\omega t
\end{align}

Las soluciones estacionarias son:
\[
  q_1 = \frac{\omega_0^2\eta_0}{\omega_0^2-\omega^2}\cos\omega t,\qquad
  q_2 = \frac{\omega_0^2\eta_0}{3\omega_0^2-\omega^2}\cos\omega t
\]
con $x_1 = \frac{1}{2}(q_1+q_2)$, $x_2 = \frac{1}{2}(q_1-q_2)$, y resonancias en
$\omega = \omega_0$ y $\omega = \omega_0\sqrt{3}$.
\end{ejemplo}

\subsection{Introducción}

\begin{definicion}[Onda]
Una \textbf{onda} es una perturbación que se propaga en el espacio y en el
tiempo, transportando energía e información sin transporte neto de materia.
\end{definicion}

Las diapositivas distinguen dos tipos básicos:
\begin{itemize}
  \item \textbf{Ondas transversales:} la oscilación del medio es perpendicular
        a la dirección de propagación.
  \item \textbf{Ondas longitudinales:} la oscilación del medio es paralela a la
        dirección de propagación.
\end{itemize}

\subsection{Propagación y ondas estacionarias}

Al reflejarse una onda en el extremo de una cuerda y superponerse con la onda
incidente, aparecen \textbf{ondas estacionarias}.

\begin{proposicion}[Superposición de dos ondas opuestas]
Si
\begin{equation}
  y_i(x,t) = A\sen(kx-\omega t),
  \label{eq:t3-incidente}
\end{equation}
y
\begin{equation}
  y_r(x,t) = A\sen(kx+\omega t),
  \label{eq:t3-reflejada}
\end{equation}
entonces la resultante es:
\begin{mdframed}[style=resultado]
\begin{equation}
  y(x,t)=y_i+y_r=2A\sen(kx)\cos(\omega t)=A'(x)\cos(\omega t).
  \label{eq:t3-estacionaria}
\end{equation}
\end{mdframed}
\end{proposicion}
\begin{variablesprimera}
  \varitem{x}{posición a lo largo de la cuerda}
  \varitem{t}{tiempo}
  \varitem{A}{amplitud de cada onda viajera}
  \varitem{k}{número de onda, $k=2\pi/\lambda$}
  \varitem{\omega}{frecuencia angular, $\omega=2\pi\nu$}
  \varitem{\lambda}{longitud de onda}
  \varitem{\nu}{frecuencia}
  \varitem{A'(x)}{amplitud espacial de la onda estacionaria, $A'(x)=2A\sen(kx)$}
\end{variablesprimera}

\begin{proposicion}[Nodos y antinodos]
Para la onda estacionaria~\eqref{eq:t3-estacionaria}:
\begin{align}
  \text{Nodos:}\quad &\sen(kx)=0
  \Longrightarrow kx=n\pi
  \Longrightarrow x_n=\frac{n\lambda}{2},
  \label{eq:t3-nodos} \\
  \text{Antinodos:}\quad &|\sen(kx)|=1
  \Longrightarrow kx=\frac{(2n+1)\pi}{2}
  \Longrightarrow x_n^{(a)}=\frac{(2n+1)\lambda}{4}.
  \label{eq:t3-antinodos}
\end{align}
\end{proposicion}
\begin{variablesprimera}
  \varitem{n}{entero no negativo que enumera nodos o antinodos}
  \varitem{x_n}{posición del nodo $n$}
  \varitem{x_n^{(a)}}{posición del antinodo $n$}
\end{variablesprimera}

\begin{proposicion}[Cuerda fija en ambos extremos]
Si una cuerda de longitud $L$ tiene nodos en $x=0$ y $x=L$, entonces:
\begin{mdframed}[style=resultado]
\begin{align}
  \lambda_n &= \frac{2L}{n}, \label{eq:t3-lambda-n}\\
  \nu_n &= \frac{nv}{2L}, \qquad
  \omega_n = \frac{n\pi v}{L},
  \qquad n=1,2,3,\dots
  \label{eq:t3-frec-n}
\end{align}
\end{mdframed}
El caso $n=1$ es la frecuencia fundamental (primer armónico).
\end{proposicion}
\begin{variablesprimera}
  \varitem{L}{longitud de la cuerda}
  \varitem{v}{velocidad de propagación en el medio}
  \varitem{\lambda_n}{longitud de onda del modo $n$}
  \varitem{\nu_n}{frecuencia del modo $n$}
  \varitem{\omega_n}{frecuencia angular del modo $n$}
\end{variablesprimera}

\subsection{Ecuación de onda en una dimensión}

\begin{definicion}[Onda viajera armónica]
Una forma general de onda viajera en una cuerda es:
\begin{equation}
  y(x,t)=A\sen(kx-\omega t+\varphi),
  \label{eq:t3-viajera}
\end{equation}
con fase inicial $\varphi$.
\end{definicion}
\begin{variablesprimera}
  \varitem{y(x,t)}{desplazamiento transversal del punto $x$ en el instante $t$}
  \varitem{\varphi}{fase inicial determinada por las condiciones iniciales}
\end{variablesprimera}

\begin{proposicion}[Velocidad de propagación]
Si dos puntos de la misma fase satisfacen
$y(x,t)=y(x+\Delta x,t+\Delta t)$, entonces:
\begin{mdframed}[style=resultado]
\begin{equation}
  v=\frac{\Delta x}{\Delta t}
   =\frac{\omega}{k}
   =\lambda\nu.
  \label{eq:t3-vel}
\end{equation}
\end{mdframed}
\end{proposicion}

\begin{proposicion}[Relaciones diferenciales para incidente y reflejada]
Para la onda incidente~\eqref{eq:t3-incidente}:
\begin{equation}
  \pdv{y_i}{x}=-\frac{1}{v}\pdv{y_i}{t}.
  \label{eq:t3-deriv-inc}
\end{equation}
Para la onda reflejada~\eqref{eq:t3-reflejada}:
\begin{equation}
  \pdv{y_r}{x}=+\frac{1}{v}\pdv{y_r}{t}.
  \label{eq:t3-deriv-ref}
\end{equation}
En ambos casos, al derivar de nuevo:
\begin{mdframed}[style=resultado]
\begin{equation}
  \pdv[2]{y}{x}=\frac{1}{v^2}\pdv[2]{y}{t}.
  \label{eq:t3-onda-1d}
\end{equation}
\end{mdframed}
\end{proposicion}

\begin{observacion}
Cuando la cuerda es muy larga respecto a la longitud de onda ($L\gg\lambda$),
los efectos de reflexión pueden despreciarse localmente y se observa
principalmente la onda incidente.
\end{observacion}

\subsection{Paquetes de onda y análisis de Fourier}

\begin{definicion}[Paquete de ondas]
Un \textbf{paquete de ondas} es una superposición lineal de ondas armónicas con
números de onda y frecuencias próximos, que produce un pulso espacialmente
localizado y con envolvente móvil.
\end{definicion}

Tomemos dos ondas cercanas:
\begin{align}
  y_1(x,t) &= A\sen\!\bigl(2\pi(k_1x-\nu_1 t)\bigr),
  \label{eq:t3-y1} \\
  y_2(x,t) &= A\sen\!\bigl(2\pi(k_2x-\nu_2 t)\bigr).
  \label{eq:t3-y2}
\end{align}
Aplicando identidades trigonométricas:
\begin{mdframed}[style=resultado]
\begin{equation}
  y(x,t)=2A\cos\!\Bigl[\pi\bigl(\Delta k\,x-\Delta\nu\,t\bigr)\Bigr]
  \sen\!\Bigl[2\pi\bigl(\bar{k}\,x-\bar{\nu}\,t\bigr)\Bigr],
  \label{eq:t3-paquete}
\end{equation}
\end{mdframed}
con
\begin{equation}
  \bar{k}=\frac{k_1+k_2}{2},\qquad
  \bar{\nu}=\frac{\nu_1+\nu_2}{2},\qquad
  \Delta k=k_2-k_1,\qquad
  \Delta \nu=\nu_2-\nu_1.
  \label{eq:t3-medias}
\end{equation}
\begin{variablesprimera}
  \varitem{k_1,k_2}{números de onda de las ondas superpuestas}
  \varitem{\nu_1,\nu_2}{frecuencias de las ondas superpuestas}
  \varitem{\bar{k},\bar{\nu}}{valores medios de número de onda y frecuencia}
  \varitem{\Delta k,\Delta\nu}{separaciones espectrales en número de onda y frecuencia}
\end{variablesprimera}

\begin{proposicion}[Expansión de Fourier]
Toda señal periódica de período $T_0=2\pi/\omega_0$ puede escribirse como:
\begin{mdframed}[style=resultado]
\begin{equation}
  y(t)=\frac{a_0}{2}+\sum_{n=1}^{\infty}
  \left[a_n\cos(n\omega_0 t)+b_n\sen(n\omega_0 t)\right].
  \label{eq:t3-fourier}
\end{equation}
\end{mdframed}
De forma equivalente:
\begin{equation}
  y(t)=\frac{a_0}{2}+\sum_{n=1}^{\infty}C_n\cos(n\omega_0 t-\delta_n).
  \label{eq:t3-fourier-fase}
\end{equation}
\end{proposicion}
\begin{variablesprimera}
  \varitem{T_0}{período fundamental de la señal}
  \varitem{\omega_0}{frecuencia angular fundamental}
  \varitem{a_n,b_n}{coeficientes trigonométricos de Fourier}
  \varitem{C_n,\delta_n}{amplitud y fase del armónico $n$}
\end{variablesprimera}

\begin{ejemplo}[Pulso periódico por tramos (diapositiva de paquetes)]
Un modelo ideal del ejemplo de diapositiva (100 ciclos del armónico
$N=1000$ en una décima del período básico) es:
\begin{equation}
  y(t)=
  \begin{cases}
    A\sen(N\omega t), & 0\le t\le \dfrac{100\pi}{N\omega},\\[6pt]
    0, & \dfrac{100\pi}{N\omega}<t\le \dfrac{\pi}{\omega},
  \end{cases}
  \qquad N=1000.
  \label{eq:t3-pulso-ej}
\end{equation}
Este tipo de señal se representa como combinación de muchos armónicos (serie
de Fourier), y su localización temporal implica ancho espectral amplio.
\end{ejemplo}

\subsection{Velocidad de fase y velocidad de grupo}

\begin{definicion}[Velocidades de fase y de grupo]
Para una relación de dispersión $\nu=\nu(k)$ (o $\omega=\omega(k)$):
\begin{mdframed}[style=resultado]
\begin{equation}
  v_f=\frac{\nu}{k}=\frac{\omega}{k},
  \qquad
  v_g=\frac{d\nu}{dk}=\frac{d\omega}{dk}.
  \label{eq:t3-vf-vg}
\end{equation}
\end{mdframed}
\end{definicion}
\begin{variablesprimera}
  \varitem{v_f}{velocidad de fase (avance de crestas)}
  \varitem{v_g}{velocidad de grupo (avance de la envolvente)}
\end{variablesprimera}

En la cuerda discreta de la sección anterior (separación $l$ entre masas), la
relación de dispersión modal puede escribirse como:
\begin{equation}
  \omega(k)=2\omega_0\sen\!\left(\frac{kl}{2}\right),
  \qquad 0\le kl\le\pi.
  \label{eq:t3-disp-discreta}
\end{equation}
\begin{variablesprimera}
  \varitem{l}{separación entre masas consecutivas en el modelo discreto}
\end{variablesprimera}
Por tanto:
\begin{align}
  v_f(k)&=\frac{\omega(k)}{k}
  =\frac{2\omega_0}{k}\sen\!\left(\frac{kl}{2}\right),
  \label{eq:t3-vf-discreta}\\
  v_g(k)&=\dv{\omega}{k}
  =\omega_0 l\cos\!\left(\frac{kl}{2}\right).
  \label{eq:t3-vg-discreta}
\end{align}

\begin{observacion}
Para $kl\ll1$, $v_f\approx v_g\approx \omega_0 l=\sqrt{T/\mu}$ y el medio es
casi no dispersivo. Para valores grandes de $k$, las ondas cortas y de mayor
frecuencia se propagan más lentamente: aparece la dispersión mostrada en clase.
\end{observacion}

\subsection{Ecuación de onda en dos y tres dimensiones}

\begin{proposicion}[Forma general]
La ecuación de onda escalar para un campo $\Psi(\mathbf{r},t)$ es:
\begin{mdframed}[style=resultado]
\begin{equation}
  \nabla^2\Psi=\frac{1}{v^2}\pdv[2]{\Psi}{t}.
  \label{eq:t3-onda-general}
\end{equation}
\end{mdframed}
\end{proposicion}
\begin{variablesprimera}
  \varitem{\Psi}{campo escalar ondulatorio (desplazamiento, presión, etc.)}
  \varitem{\mathbf{r}}{vector posición del punto del medio}
  \varitem{r,\theta,\phi}{coordenadas esféricas: radio, ángulo polar y ángulo azimutal}
\end{variablesprimera}

\paragraph{En coordenadas cartesianas (2D y 3D).}
\begin{align}
  \pdv[2]{\Psi}{x}+\pdv[2]{\Psi}{y}
  &=\frac{1}{v^2}\pdv[2]{\Psi}{t}
  \label{eq:t3-onda-2d-cart} \\
  \pdv[2]{\Psi}{x}+\pdv[2]{\Psi}{y}+\pdv[2]{\Psi}{z}
  &=\frac{1}{v^2}\pdv[2]{\Psi}{t}.
  \label{eq:t3-onda-3d-cart}
\end{align}

\paragraph{En coordenadas cilíndricas (simetría axial, sin dependencia en $\phi$).}
\begin{equation}
  \pdv[2]{\Psi}{r}+\frac{1}{r}\pdv{\Psi}{r}
  =\frac{1}{v^2}\pdv[2]{\Psi}{t}.
  \label{eq:t3-onda-cil}
\end{equation}

\paragraph{En coordenadas esféricas (3D completo).}
\begin{equation}
  \frac{1}{r^2}\pdv{}{r}\!\left(r^2\pdv{\Psi}{r}\right)
  +\frac{1}{r^2\sen\theta}\pdv{}{\theta}\!\left(\sen\theta\pdv{\Psi}{\theta}\right)
  +\frac{1}{r^2\sen^2\theta}\pdv[2]{\Psi}{\phi}
  =\frac{1}{v^2}\pdv[2]{\Psi}{t}.
  \label{eq:t3-onda-esf}
\end{equation}

\subsection{Interferencia y difracción}

\begin{definicion}[Interferencia]
La \textbf{interferencia} es el fenómeno por el cual varias ondas coinciden en
una región del espacio y se combinan por superposición. Tras la interacción,
cada onda prosigue su propagación sin perder identidad.
\end{definicion}

\begin{proposicion}[Condiciones de interferencia de dos focos]
Si $\Delta r=|r_2-r_1|$ es la diferencia de caminos desde dos fuentes
coherentes:
\begin{mdframed}[style=resultado]
\begin{align}
  \text{Interferencia constructiva:}\quad &\Delta r=n\lambda,
  \label{eq:t3-constructiva}\\
  \text{Interferencia destructiva:}\quad &\Delta r=\left(2n+1\right)\frac{\lambda}{2}.
  \label{eq:t3-destructiva}
\end{align}
\end{mdframed}
\end{proposicion}
\begin{variablesprimera}
  \varitem{r_1,r_2}{distancias desde cada foco al punto de observación}
  \varitem{\Delta r}{diferencia de caminos ópticos/geométricos}
\end{variablesprimera}

\subsection{Ejemplos de las presentaciones}

\begin{ejemplo}[Lectura de ecuación de onda desde una figura]
En la diapositiva aparece el enunciado ``Escriba la ecuación de la onda que
aparece en la figura''. El procedimiento general (correcto y completo) es:
\begin{enumerate}
  \item Leer amplitud $A$, longitud de onda $\lambda$ y período $T$ (o
        frecuencia $\nu$).
  \item Calcular $k=2\pi/\lambda$ y $\omega=2\pi/T=2\pi\nu$.
  \item Elegir el signo según el sentido de propagación:
        $y(x,t)=A\sen(kx-\omega t+\varphi)$ (hacia $+x$) o
        $y(x,t)=A\sen(kx+\omega t+\varphi)$ (hacia $-x$).
  \item Ajustar la fase $\varphi$ con la condición inicial dada en el dibujo.
\end{enumerate}
\end{ejemplo}

\begin{ejemplo}[Onda transversal dada explícitamente]
Sea
\begin{equation}
  y(x,t)=0.3\,\sen\!\bigl[\pi(0.5x-50t)\bigr],
  \label{eq:t3-ej-onda}
\end{equation}
con $x,y$ en cm y $t$ en s.
\begin{itemize}
  \item Amplitud: $A=0.3\,\mathrm{cm}$.
  \item Número de onda: $k=0.5\pi\,\mathrm{cm^{-1}}$.
  \item Frecuencia angular: $\omega=50\pi\,\mathrm{s^{-1}}$.
  \item Longitud de onda: $\lambda=2\pi/k=4\,\mathrm{cm}$.
  \item Frecuencia: $\nu=\omega/(2\pi)=25\,\mathrm{Hz}$.
  \item Período: $T=1/\nu=0.04\,\mathrm{s}$.
  \item Velocidad de propagación: $v=\omega/k=100\,\mathrm{cm\,s^{-1}}=1\,\mathrm{m\,s^{-1}}$.
\end{itemize}
La velocidad transversal máxima de una partícula es:
\begin{equation}
  v_{y,\max}=A\omega=0.3\cdot50\pi=15\pi\,\mathrm{cm\,s^{-1}}
  \approx 47.1\,\mathrm{cm\,s^{-1}}.
  \label{eq:t3-vy-max}
\end{equation}
\end{ejemplo}

\begin{ejemplo}[Cuerda larga forzada en $x=0$]
Datos: $\mu=0.1\,\mathrm{kg\,m^{-1}}$, $T=50\,\mathrm{N}$, amplitud en el
extremo impulsor $A=2\,\mathrm{cm}=0.02\,\mathrm{m}$, período
$\tau=0.1\,\mathrm{s}$.

\begin{align}
  v&=\sqrt{\frac{T}{\mu}}=\sqrt{500}\approx22.36\,\mathrm{m\,s^{-1}},
  \label{eq:t3-ej-v}\\
  \nu&=\frac{1}{\tau}=10\,\mathrm{Hz},\qquad
  \lambda=\frac{v}{\nu}\approx2.236\,\mathrm{m},
  \label{eq:t3-ej-lambda}
\end{align}
y por tanto
\begin{equation}
  \omega=2\pi\nu=20\pi\,\mathrm{rad\,s^{-1}},\qquad
  k=\frac{2\pi}{\lambda}\approx2.810\,\mathrm{rad\,m^{-1}}.
  \label{eq:t3-ej-komega}
\end{equation}

La onda hacia $+x$ puede escribirse como
$y(x,t)=A\sen(kx-\omega t+\varphi)$. Si $y(0,0)=0.01\,\mathrm{m}$:
\begin{equation}
  A\sen\varphi=0.01\;\Longrightarrow\;\sen\varphi=0.5.
  \label{eq:t3-ej-fase1}
\end{equation}
Tomando la condición de velocidad transversal inicial negativa en $x=0$,
$\dot y(0,0)=-A\omega\cos\varphi<0$, resulta $\cos\varphi>0$, luego
$\varphi=\pi/6$.
\begin{mdframed}[style=resultado]
\begin{equation}
  y(x,t)=0.02\,\sen\!\left(2.810\,x-20\pi\,t+\frac{\pi}{6}\right)\;\mathrm{m}.
  \label{eq:t3-ej-cuerda-final}
\end{equation}
\end{mdframed}
\end{ejemplo}

\begin{ejemplo}[Paquete inicial triangular en cuerda fija]
Para una cuerda de longitud $L$ con extremos fijos y condición inicial
triangular de altura $h$ en el punto $x=a$:
\begin{equation}
  y(x,0)=
  \begin{cases}
    \dfrac{h}{a}\,x, & 0\le x\le a,\\[6pt]
    \dfrac{h}{L-a}(L-x), & a\le x\le L,
  \end{cases}
  \qquad
  \pdv{y}{t}(x,0)=0.
  \label{eq:t3-ej-triang}
\end{equation}
La solución modal es:
\begin{equation}
  y(x,t)=\sum_{n=1}^{\infty} b_n
  \sen\!\left(\frac{n\pi x}{L}\right)
  \cos\!\left(\frac{n\pi vt}{L}\right),
  \label{eq:t3-ej-serie}
\end{equation}
con
\begin{equation}
  b_n=\frac{2hL^2}{a(L-a)\pi^2n^2}
  \sen\!\left(\frac{n\pi a}{L}\right).
  \label{eq:t3-ej-bn}
\end{equation}
Si el pico está en el centro ($a=L/2$), solo sobreviven armónicos impares:
\begin{equation}
  b_n=
  \begin{cases}
    \dfrac{8h}{\pi^2n^2}, & n\text{ impar},\\[6pt]
    0, & n\text{ par}.
  \end{cases}
  \label{eq:t3-ej-centro}
\end{equation}
\end{ejemplo}

\begin{ejemplo}[Seis partículas idénticas entre extremos fijos]
En el modelo discreto de cuerda vibrante con $N=6$ masas iguales:
\begin{align}
  A_p^{(n)}&=C_n\sen\!\left(\frac{n\,p\,\pi}{7}\right),
  \qquad p=1,\dots,6,
  \label{eq:t3-ej6-perfil}\\
  \omega_n&=2\omega_0\sen\!\left(\frac{n\pi}{14}\right),
  \qquad n=1,\dots,6.
  \label{eq:t3-ej6-frec}
\end{align}
Los seis modos normales son precisamente esos seis valores de $n$, con
$n-1$ nodos internos para el modo $n$.
\end{ejemplo}

\begin{ejemplo}[Ondas capilares-gravitatorias en agua profunda]
La relación de dispersión dada en la diapositiva es:
\begin{equation}
  \omega^2(k)=gk+\frac{T}{\rho}k^3.
  \label{eq:t3-ej-disp-agua}
\end{equation}
Por tanto,
\begin{align}
  v_f(k)&=\frac{\omega}{k}
  =\sqrt{\frac{g}{k}+\frac{T}{\rho}k},
  \label{eq:t3-ej-vf}\\
  v_g(k)&=\dv{\omega}{k}
  =\frac{g+3(T/\rho)k^2}{2\sqrt{gk+(T/\rho)k^3}}.
  \label{eq:t3-ej-vg}
\end{align}
En función de $\lambda$ (con $k=2\pi/\lambda$):
\begin{equation}
  v_f(\lambda)=
  \sqrt{\frac{g\lambda}{2\pi}+\frac{2\pi T}{\rho\,\lambda}}.
  \label{eq:t3-ej-vf-lambda}
\end{equation}
El mínimo de $v_f$ se obtiene de $\dv{}{k}(v_f^2)=0$:
\begin{equation}
  k_\star=\sqrt{\frac{\rho g}{T}},
  \qquad
  \lambda_\star=2\pi\sqrt{\frac{T}{\rho g}}.
  \label{eq:t3-ej-lambda-star}
\end{equation}
Con $T=7.2\times10^{-2}\,\mathrm{N\,m^{-1}}$, $\rho=997\,\mathrm{kg\,m^{-3}}$:
\begin{equation}
  \lambda_\star\approx1.70\times10^{-2}\,\mathrm{m}=1.70\,\mathrm{cm},
  \qquad
  v_{f,\min}\approx0.231\,\mathrm{m\,s^{-1}}.
  \label{eq:t3-ej-num}
\end{equation}
\end{ejemplo}

\begin{ejemplo}[Modelo lineal de la molécula de CO$_2$]
Modelamos CO$_2$ como tres masas alineadas con dos muelles idénticos de
constante $k$: extremos $m_1=m_3=m_O$ y centro $m_2=m_C$.
Las ecuaciones longitudinales son:
\begin{align}
  m_O\ddot{x}_1&=-k(x_1-x_2), \label{eq:t3-co2-1}\\
  m_C\ddot{x}_2&=-k(2x_2-x_1-x_3), \label{eq:t3-co2-2}\\
  m_O\ddot{x}_3&=-k(x_3-x_2). \label{eq:t3-co2-3}
\end{align}
Con ansatz armónico $x_j=A_j\cos\omega t$ aparecen dos modos vibracionales
independientes (además del modo traslacional $\omega=0$):
\begin{itemize}
  \item \textbf{Modo antisimétrico} ($A_1=-A_3$, $A_2=0$):
        \[
          \omega_a^2=\frac{k}{m_O}.
        \]
  \item \textbf{Modo simétrico de estiramiento} ($A_1=A_3$):
        \[
          \omega_s^2=\frac{k}{m_O}+\frac{2k}{m_C}.
        \]
\end{itemize}
\end{ejemplo}

\section{Tema 3: Elasticidad y Mecánica de Fluidos}

\subsection{Introducción a la teoría de la elasticidad}

\begin{definicion}[Elasticidad]
La \textbf{elasticidad} estudia cómo un sólido se deforma cuando actúan
fuerzas externas y cómo recupera su forma al retirarlas, dentro del régimen
elástico.
\end{definicion}

Las diapositivas destacan aplicaciones en estructuras, ingeniería de
materiales, biomecánica y geofísica.

\subsection{Esfuerzo y deformación}

\begin{definicion}[Esfuerzo]
El esfuerzo es la fuerza por unidad de área:
\begin{equation}
  \sigma=\frac{F}{A},
  \qquad [\sigma]=\mathrm{Pa}.
  \label{eq:t4-esfuerzo}
\end{equation}
Para esfuerzos tangenciales se usa $\tau$ (esfuerzo cortante).
\end{definicion}
\begin{variablesprimera}
  \varitem{F}{fuerza aplicada sobre una superficie}
  \varitem{A}{área sobre la que actúa la fuerza}
  \varitem{\sigma}{esfuerzo normal}
  \varitem{\tau}{esfuerzo cortante}
\end{variablesprimera}

\begin{definicion}[Deformación]
La deformación lineal (ingenieril) y la deformación por cizalla se escriben:
\begin{equation}
  \epsilon=\frac{\Delta l}{l},
  \qquad
  \gamma \approx \tan\theta \approx \theta.
  \label{eq:t4-def}
\end{equation}
\end{definicion}
\begin{variablesprimera}
  \varitem{\Delta l}{cambio de longitud}
  \varitem{l}{longitud inicial}
  \varitem{\epsilon}{deformación lineal relativa}
  \varitem{\gamma}{deformación angular de cizalla}
  \varitem{\theta}{ángulo de distorsión (pequeño)}
\end{variablesprimera}

\begin{proposicion}[Ley de Hooke uniaxial y de cizalla]
En el régimen lineal:
\begin{mdframed}[style=resultado]
\begin{align}
  \sigma&=E\,\epsilon, \label{eq:t4-hooke-uni}\\
  \tau&=G\,\gamma. \label{eq:t4-hooke-cort}
\end{align}
\end{mdframed}
\end{proposicion}
\begin{variablesprimera}
  \varitem{E}{módulo de Young}
  \varitem{G}{módulo de cizalla}
\end{variablesprimera}

\begin{nota}
Tal como se comenta en las diapositivas, en materiales no lineales la curva
$\sigma$--$\epsilon$ no es recta y se emplean \emph{módulo secante} y
\emph{módulo tangente} para caracterizar la rigidez local.
\end{nota}

\begin{observacion}
Como en las diapositivas, distinguimos deformación \textbf{elástica}
(reversible) y \textbf{plástica} (irreversible) separadas por el límite elástico
$\sigma_y$.
\end{observacion}

\subsection{Tensor de esfuerzos}

\begin{definicion}[Vector de tracción y tensor de Cauchy]
Sobre un elemento de área $dA$ con normal unitaria $\mathbf{n}$:
\begin{equation}
  d\mathbf{F}=\boldsymbol{\Sigma}\,\mathbf{n}\,dA,
  \qquad
  \mathbf{t}^{(\mathbf{n})}=\frac{d\mathbf{F}}{dA}=\boldsymbol{\Sigma}\,\mathbf{n}.
  \label{eq:t4-traccion}
\end{equation}
Aquí $\boldsymbol{\Sigma}$ es el tensor de esfuerzos (tensor de Cauchy).
\end{definicion}
\begin{variablesprimera}
  \varitem{dA}{área diferencial de una superficie interna}
  \varitem{\mathbf{n}}{normal unitaria a la superficie}
  \varitem{d\mathbf{F}}{fuerza interna sobre el elemento de área}
  \varitem{\mathbf{t}^{(\mathbf{n})}}{vector tensión o tracción}
  \varitem{\boldsymbol{\Sigma}}{tensor de esfuerzos}
\end{variablesprimera}

En componentes cartesianas:
\begin{equation}
  \boldsymbol{\Sigma}=
  \begin{pmatrix}
    \sigma_{xx} & \tau_{xy} & \tau_{xz}\\
    \tau_{yx} & \sigma_{yy} & \tau_{yz}\\
    \tau_{zx} & \tau_{zy} & \sigma_{zz}
  \end{pmatrix},
  \qquad
  \tau_{ij}=\tau_{ji}.
  \label{eq:t4-sigma-mat}
\end{equation}

\begin{ejemplo}[Tensor de esfuerzos y tracción sobre un plano]
En el punto $P$ se da
\begin{equation}
  \boldsymbol{\Sigma}=
  \begin{pmatrix}
    1 & 2 & 0\\
    2 & 2 & 0\\
    0 & 0 & 1
  \end{pmatrix}.
  \label{eq:t4-ej-sigma}
\end{equation}
Para un elemento paralelo al plano $x+y+z=0$ en el octante positivo,
tomamos la normal exterior
\begin{equation}
  \mathbf{n}=\frac{1}{\sqrt{3}}(1,1,1).
  \label{eq:t4-ej-normal}
\end{equation}
La tracción es
\begin{equation}
  \mathbf{t}^{(\mathbf{n})}=\boldsymbol{\Sigma}\mathbf{n}
  =\frac{1}{\sqrt{3}}(3,4,1),
  \qquad
  d\mathbf{F}=\mathbf{t}^{(\mathbf{n})}\,dA.
  \label{eq:t4-ej-trac}
\end{equation}
El ángulo $\theta$ entre fuerza y normal:
\begin{equation}
  \cos\theta=
  \frac{\mathbf{t}^{(\mathbf{n})}\cdot\mathbf{n}}
       {\|\mathbf{t}^{(\mathbf{n})}\|}
  =\frac{8}{\sqrt{78}}
  \Longrightarrow
  \theta\approx25.0^\circ.
  \label{eq:t4-ej-angulo}
\end{equation}
\end{ejemplo}

\subsection{Tensor de deformación}

Sea $\mathbf{u}(\mathbf{r})$ el campo de desplazamientos. La variación
infinitesimal es:
\begin{equation}
  d\mathbf{u}=\mat{D}\,d\mathbf{r},
  \qquad
  D_{ij}=\pdv{u_i}{x_j}.
  \label{eq:t4-grad-u}
\end{equation}
\begin{variablesprimera}
  \varitem{\mathbf{u}(\mathbf{r})}{desplazamiento de cada punto del sólido}
  \varitem{\mat{D}}{tensor gradiente del desplazamiento}
  \varitem{u_i}{componente $i$ del desplazamiento}
  \varitem{x_j}{coordenada espacial $j$}
\end{variablesprimera}

La descomposición estándar es:
\begin{equation}
  \mat{D}=\mat{A}+\mat{E},
  \qquad
  \mat{A}=\frac{\mat{D}-\mat{D}'}{2},
  \qquad
  \mat{E}=\frac{\mat{D}+\mat{D}'}{2}.
  \label{eq:t4-D-A-E}
\end{equation}
\begin{itemize}
  \item $\mat{A}$ (antisimétrica) representa rotación rígida local.
  \item $\mat{E}$ (simétrica) es el tensor de deformación.
\end{itemize}

Además,
\begin{equation}
  \mat{E}=\frac{e}{3}\mat{I}+\mat{E}',
  \qquad
  e=\tr(\mat{E}),
  \qquad
  \tr(\mat{E}')=0,
  \label{eq:t4-esferica-desv}
\end{equation}
donde $\frac{e}{3}\mat{I}$ es la parte esférica y $\mat{E}'$ la desviadora.
\begin{variablesprimera}
  \varitem{\mat{A}}{parte antisimétrica del gradiente, asociada a rotación}
  \varitem{\mat{E}}{tensor de deformación infinitesimal}
  \varitem{e}{deformación volumétrica infinitesimal (traza de $\mat{E}$)}
  \varitem{\mat{E}'}{parte desviadora de la deformación}
\end{variablesprimera}

\begin{ejemplo}[Descomposición esférica y desviadora]
Si
\begin{equation}
  \mat{E}=
  \begin{pmatrix}
    0.01 & 0.02 & 0.05\\
    0.02 & 0.03 & 0.04\\
    0.05 & 0.04 & 0.04
  \end{pmatrix},
  \label{eq:t4-ej-E}
\end{equation}
entonces
\begin{equation}
  e=\tr(\mat{E})=0.08,\qquad
  \frac{e}{3}\mat{I}=0.026666\ldots\,\mat{I}.
  \label{eq:t4-ej-e}
\end{equation}
La parte desviadora vale:
\begin{equation}
  \mat{E}'=
  \begin{pmatrix}
   -0.0167 & 0.0200 & 0.0500\\
    0.0200 & 0.0033 & 0.0400\\
    0.0500 & 0.0400 & 0.0133
  \end{pmatrix},
  \qquad \tr(\mat{E}')=0.
  \label{eq:t4-ej-Edev}
\end{equation}
\end{ejemplo}

\begin{ejemplo}[Tensor de cizalla pura]
Para
\begin{equation}
  \mat{E}=
  \begin{pmatrix}
    0 & \gamma & 0\\
    \gamma & 0 & 0\\
    0 & 0 & 0
  \end{pmatrix},
  \label{eq:t4-ej-cizalla}
\end{equation}
se trata de una cizalla pura en el plano $xy$: no hay deformación
volumétrica ($e=0$) y sí distorsión angular proporcional a $\gamma$.
\end{ejemplo}

\subsection{Ley de Hooke generalizada y ecuación de Navier (sólidos)}

Para material isótropo lineal:
\begin{equation}
  \boldsymbol{\Sigma}=K\,e\,\mat{I}+2G\,\mat{E}'
  =\lambda\,e\,\mat{I}+2G\,\mat{E}.
  \label{eq:t4-hooke-general}
\end{equation}
\begin{equation}
  \lambda=K-\frac{2G}{3},
  \qquad
  E=\frac{9KG}{3K+G}.
  \label{eq:t4-modulos}
\end{equation}
\begin{variablesprimera}
  \varitem{K}{módulo volumétrico o de compresibilidad (MCo)}
  \varitem{\lambda}{primera constante de Lamé}
  \varitem{E}{módulo de Young}
\end{variablesprimera}

El balance de cantidad de movimiento en un sólido elástico da:
\begin{equation}
  \rho\,\ddot{\mathbf{u}}=\rho\,\mathbf{g}+\nabla\cdot\boldsymbol{\Sigma}.
  \label{eq:t4-balance-solido}
\end{equation}
Sustituyendo~\eqref{eq:t4-hooke-general}:
\begin{mdframed}[style=resultado]
\begin{equation}
  \rho\,\ddot{\mathbf{u}}=
  \rho\,\mathbf{g}+
  \left(K+\frac{G}{3}\right)\nabla(\nabla\cdot\mathbf{u})+
  G\,\nabla^2\mathbf{u}.
  \label{eq:t4-navier-solido}
\end{equation}
\end{mdframed}
\begin{variablesprimera}
  \varitem{\rho}{densidad de masa del sólido}
  \varitem{\mathbf{g}}{aceleración de la gravedad (u otra fuerza de cuerpo)}
\end{variablesprimera}

Velocidades de ondas elásticas en medio isótropo:
\begin{align}
  c_L&=\sqrt{\frac{K+\frac{4}{3}G}{\rho}},
  \label{eq:t4-cL}\\
  c_T&=\sqrt{\frac{G}{\rho}}.
  \label{eq:t4-cT}
\end{align}

\begin{ejemplo}[Granito: velocidades longitudinal y transversal]
Con $K\approx40\,\mathrm{GPa}$, $G\approx25\,\mathrm{GPa}$ y
$\rho\approx2.7\times10^3\,\mathrm{kg\,m^{-3}}$:
\begin{equation}
  c_T\approx3.04\times10^3\,\mathrm{m\,s^{-1}},
  \qquad
  c_L\approx5.21\times10^3\,\mathrm{m\,s^{-1}}.
  \label{eq:t4-granito}
\end{equation}
\end{ejemplo}

\subsection{Introducción a la mecánica de fluidos}

\begin{definicion}[Fluido]
Un fluido es una sustancia (líquido o gas) que se deforma continuamente
bajo cualquier esfuerzo cortante no nulo.
\end{definicion}

La mecánica de fluidos se divide, como en la presentación, en:
\begin{itemize}
  \item \textbf{Hidrostática:} fluidos en reposo.
  \item \textbf{Hidrodinámica:} fluidos en movimiento.
\end{itemize}

\begin{definicion}[Fluido ideal no viscoso]
En el modelo de fluido ideal (sin viscosidad) el tensor de esfuerzos se reduce
a presión isotrópica:
\begin{equation}
  \boldsymbol{\Sigma}=-p\,\mat{I}.
  \label{eq:t4-sigma-fluido-ideal}
\end{equation}
\end{definicion}
\begin{variablesprimera}
  \varitem{p}{presión termodinámica local del fluido}
\end{variablesprimera}

Entonces, el balance de cantidad de movimiento se escribe como ecuación de
Euler:
\begin{mdframed}[style=resultado]
\begin{equation}
  \rho\,\frac{D\mathbf{v}}{Dt}
  =\rho\,\mathbf{g}-\nabla p.
  \label{eq:t4-euler}
\end{equation}
\end{mdframed}
\begin{variablesprimera}
  \varitem{\mathbf{v}}{campo de velocidad del fluido}
  \varitem{\frac{D}{Dt}}{derivada material, $\frac{\partial}{\partial t}+\mathbf{v}\cdot\nabla$}
\end{variablesprimera}

Para flujo estacionario, incompresible y no viscoso a lo largo de una línea
de corriente:
\begin{equation}
  p+\frac{1}{2}\rho v^2+\rho gz=\text{cte}.
  \label{eq:t4-bernoulli}
\end{equation}

\subsection{Hipótesis de continuo y ecuación de continuidad}

La conservación de masa da:
\begin{equation}
  \frac{d}{dt}\!\left(\rho\,dV\right)=0
  \Longrightarrow
  \frac{D\rho}{Dt}+\rho\,\nabla\cdot\mathbf{v}=0.
  \label{eq:t4-cont-material}
\end{equation}
Forma euleriana equivalente:
\begin{equation}
  \pdv{\rho}{t}+\nabla\cdot(\rho\mathbf{v})=0.
  \label{eq:t4-cont}
\end{equation}
Si $\rho=\mathrm{cte}$:
\begin{equation}
  \nabla\cdot\mathbf{v}=0.
  \label{eq:t4-incomp}
\end{equation}

\subsection{Pequeñas perturbaciones en fluidos (ondas acústicas)}

Partiendo de un equilibrio hidrostático y perturbando
$p=p_0+p'$, $\rho=\rho_0+\rho'$, con $|p'|,|\rho'|$ pequeños, se obtiene:
\begin{align}
  \rho_0\pdv{\mathbf{v}}{t}&=-\nabla p',
  \label{eq:t4-acu1}\\
  \pdv{\rho'}{t}+\rho_0\nabla\cdot\mathbf{v}&=0.
  \label{eq:t4-acu2}
\end{align}
Si $p'=c_s^2\rho'$ (compresibilidad lineal), resulta:
\begin{equation}
  \pdv[2]{p'}{t}=c_s^2\nabla^2p',
  \qquad
  c_s=\sqrt{\frac{K}{\rho_0}}.
  \label{eq:t4-onda-acustica}
\end{equation}
\begin{variablesprimera}
  \varitem{p',\rho'}{perturbaciones de presión y densidad respecto al equilibrio}
  \varitem{\rho_0,p_0}{valores de equilibrio}
  \varitem{c_s}{velocidad del sonido en el fluido}
\end{variablesprimera}

\subsection{Fluidos reales: viscosidad y Navier--Stokes}

Para fluido newtoniano:
\begin{equation}
  \tau=\mu\,\pdv{v_t}{n},
  \label{eq:t4-newton-visc}
\end{equation}
donde $\mu$ es la viscosidad dinámica y $v_t$ la componente tangencial.

\begin{itemize}
  \item Flujo \textbf{laminar}: capas ordenadas.
  \item Flujo \textbf{turbulento}: movimiento desordenado.
  \item Condición de no deslizamiento: en pared sólida, $\mathbf{v}=0$.
\end{itemize}

Para fluido incompresible con $\mu$ constante:
\begin{mdframed}[style=resultado]
\begin{equation}
  \rho\,\frac{D\mathbf{v}}{Dt}
  =\rho\,\mathbf{g}-\nabla p+\mu\nabla^2\mathbf{v}.
  \label{eq:t4-navier-stokes}
\end{equation}
\end{mdframed}
\begin{variablesprimera}
  \varitem{\mu}{viscosidad dinámica del fluido}

\end{variablesprimera}

\subsection{Ejemplos de las diapositivas de Elasticidad y Fluidos}

\begin{ejemplo}[Tubería que reduce su diámetro a la mitad]
Datos de diapositiva: agua con $v_1=4\,\mathrm{m\,s^{-1}}$ y
$p_1=200\,\mathrm{kPa}$; el diámetro se reduce a la mitad.

Por continuidad:
\begin{equation}
  A_1v_1=A_2v_2,\qquad
  A_2=\frac{A_1}{4}
  \Longrightarrow v_2=4v_1=16\,\mathrm{m\,s^{-1}}.
  \label{eq:t4-ej-tuberia-v}
\end{equation}
Tomando tubería horizontal y $\rho\simeq1000\,\mathrm{kg\,m^{-3}}$:
\begin{equation}
  p_2=p_1+\frac{1}{2}\rho\left(v_1^2-v_2^2\right)
  =200\,\mathrm{kPa}-120\,\mathrm{kPa}
  =80\,\mathrm{kPa}.
  \label{eq:t4-ej-tuberia-p}
\end{equation}
\end{ejemplo}

\begin{ejemplo}[Flujo gravitacional laminar en plano inclinado]
Para una película de espesor uniforme $a$ sobre un plano inclinado ángulo
$\phi$, en régimen estacionario, incompresible y unidireccional:
\begin{equation}
  g\sen\phi+\nu\,\dv[2]{v_x}{y}=0,
  \label{eq:t4-ej-plano-ode}
\end{equation}
donde $\nu=\mu/\rho$ es la viscosidad cinemática.

Con condiciones de borde $v_x(0)=0$ (no deslizamiento) y
$\tau_{xy}(a)=\mu\,dv_x/dy|_{y=a}=0$ (superficie libre):
\begin{mdframed}[style=resultado]
\begin{equation}
  v_x(y)=\frac{g\sen\phi}{\nu}\left(ay-\frac{y^2}{2}\right).
  \label{eq:t4-ej-plano-sol}
\end{equation}
\end{mdframed}
\end{ejemplo}

\begin{ejemplo}[Campo de velocidades 2D estacionario]
Sea
\begin{equation}
  \mathbf{v}(x,y)=\bigl(ax+b\bigr)\,\uvec{i}+\bigl(-ay+cx\bigr)\,\uvec{j}.
  \label{eq:t4-ej-campo-v}
\end{equation}
Es incompresible porque
$\partial v_x/\partial x+\partial v_y/\partial y=a-a=0$.

Si despreciamos viscosidad y fuerzas de cuerpo, Euler estacionaria:
\begin{equation}
  \rho(\mathbf{v}\cdot\nabla)\mathbf{v}=-\nabla p.
  \label{eq:t4-ej-euler}
\end{equation}
Se obtiene:
\begin{equation}
  p(x,y)=C-\rho\left(
    \frac{a^2}{2}x^2+ab\,x+bc\,y+\frac{a^2}{2}y^2
  \right).
  \label{eq:t4-ej-presion}
\end{equation}
\end{ejemplo}

\begin{ejemplo}[Sistema de depósito (diapositiva 3.4, resuelto)]
De la figura: desnivel total entre superficie libre y salida
$H=2.4+3.6=6.0\,\mathrm{m}$, tramo principal de diámetro
$D_A=150\,\mathrm{mm}$ y salida de diámetro $D_s=50\,\mathrm{mm}$.
Se asume flujo ideal y descarga a la atmósfera.

Entre superficie libre (0) y salida (s), Bernoulli con $p_0=p_s=p_{\text{atm}}$
y $v_0\approx0$:
\begin{equation}
  v_s=\sqrt{2gH}
  =\sqrt{2\cdot9.81\cdot6.0}
  \approx 10.85\,\mathrm{m\,s^{-1}}.
  \label{eq:t4-ej-deposito-vs}
\end{equation}

Por continuidad:
\begin{equation}
  Q=A_s v_s=A_A v_A
  \Longrightarrow
  v_A=v_s\left(\frac{D_s}{D_A}\right)^2
  =10.85\left(\frac{0.05}{0.15}\right)^2
  \approx1.21\,\mathrm{m\,s^{-1}}.
  \label{eq:t4-ej-deposito-va}
\end{equation}

Como $A$ y la salida están aproximadamente a la misma cota, Bernoulli entre
$A$ y salida:
\begin{equation}
  p_A+\frac{1}{2}\rho v_A^2
  =p_{\text{atm}}+\frac{1}{2}\rho v_s^2
  \Longrightarrow
  p_A-p_{\text{atm}}
  =\frac{1}{2}\rho\left(v_s^2-v_A^2\right).
  \label{eq:t4-ej-deposito-pa-base}
\end{equation}
Con $\rho=1000\,\mathrm{kg\,m^{-3}}$:
\begin{equation}
  p_A-p_{\text{atm}}
  \approx 5.8\times10^4\,\mathrm{Pa}
  =58\,\mathrm{kPa}\quad\text{(presión manométrica).}
  \label{eq:t4-ej-deposito-pa}
\end{equation}
\end{ejemplo}

\section{Tema 4: Relatividad Especial}

\subsection{Contexto y postulados}

Como en la presentación, la relatividad especial extiende el principio de
relatividad a \emph{toda} la física, en particular al electromagnetismo y a la
dinámica de partículas rápidas.

\begin{definicion}[Postulados de Einstein]
\begin{enumerate}
  \item Las leyes de la física son las mismas en todos los sistemas inerciales.
  \item La velocidad de la luz en vacío $c$ es la misma para todos los
        observadores inerciales.
\end{enumerate}
\end{definicion}
\begin{variablesprimera}
  \varitem{c}{velocidad de la luz en el vacío}
\end{variablesprimera}

\begin{proposicion}[Factor de Lorentz]
\begin{mdframed}[style=resultado]
\begin{equation}
  \gamma(v)=\frac{1}{\sqrt{1-\dfrac{v^2}{c^2}}}.
  \label{eq:t5-gamma}
\end{equation}
\end{mdframed}
\end{proposicion}

Consecuencias cinemáticas básicas (citadas en clase):
\begin{align}
  \Delta t&=\gamma\,\Delta\tau
  &&\text{(dilatación temporal),} \label{eq:t5-tiempo}\\
  L&=\frac{L_0}{\gamma}
  &&\text{(contracción longitudinal).} \label{eq:t5-long}
\end{align}
\begin{variablesprimera}
  \varitem{\Delta\tau}{tiempo propio}
  \varitem{\Delta t}{tiempo medido desde un sistema donde el reloj se mueve}
  \varitem{L_0}{longitud propia}
  \varitem{L}{longitud medida en el sistema donde el objeto se mueve}
\end{variablesprimera}

\subsection{Dinámica relativista}

La cantidad de movimiento relativista es:
\begin{equation}
  \mathbf{p}=\gamma m\mathbf{v}.
  \label{eq:t5-momento}
\end{equation}

La energía total y la energía cinética:
\begin{align}
  E&=\gamma mc^2, \label{eq:t5-E}\\
  E_0&=mc^2, \label{eq:t5-E0}\\
  K&=E-E_0=(\gamma-1)mc^2. \label{eq:t5-K}
\end{align}
\begin{variablesprimera}
  \varitem{m}{masa en reposo}
  \varitem{\mathbf{p}}{momento lineal relativista}
  \varitem{E}{energía total}
  \varitem{E_0}{energía en reposo}
  \varitem{K}{energía cinética}
\end{variablesprimera}

Relación energía--momento:
\begin{equation}
  E^2=(pc)^2+(mc^2)^2.
  \label{eq:t5-Ep}
\end{equation}
Para fotones ($m=0$):
\begin{equation}
  E=pc=h\nu.
  \label{eq:t5-foton}
\end{equation}
\begin{variablesprimera}
  \varitem{h}{constante de Planck}
\end{variablesprimera}

\subsection{Límite de velocidad y validez de la nueva dinámica}

De~\eqref{eq:t5-K}, la velocidad en función de $K$ es:
\begin{equation}
  v=c\sqrt{1-\frac{1}{\left(1+\dfrac{K}{mc^2}\right)^2}}<c.
  \label{eq:t5-vK}
\end{equation}
Así, al aumentar $K$, $v$ se aproxima a $c$ pero nunca la supera.

\subsection{Movimiento bajo fuerza constante colineal}

Si $\mathbf{F}$ es constante y paralela al movimiento, con partícula partiendo
del reposo:
\begin{equation}
  p(t)=Ft.
  \label{eq:t5-pFt}
\end{equation}
Combinando con~\eqref{eq:t5-momento}:
\begin{mdframed}[style=resultado]
\begin{equation}
  v(t)=\frac{Ft/m}{\sqrt{1+\left(\dfrac{Ft}{mc}\right)^2}}
      =\frac{c}{\sqrt{1+\left(\dfrac{mc}{Ft}\right)^2}}.
  \label{eq:t5-vt}
\end{equation}
\end{mdframed}
Límites:
\begin{align}
  Ft\ll mc&\Longrightarrow v(t)\approx \frac{F}{m}t
  &&\text{(newtoniano),} \label{eq:t5-lim1}\\
  Ft\gg mc&\Longrightarrow v(t)\to c
  &&\text{(régimen relativista).} \label{eq:t5-lim2}
\end{align}

\subsection{Notas de interpretación física}

\begin{itemize}
  \item El “aumento de masa” de formulaciones antiguas se interpreta hoy como
        aumento de energía y momento con $v$, manteniendo la masa en reposo
        constante.
  \item La conservación de energía y la de masa quedan unificadas mediante
        $E=mc^2$, tal como enfatizan las diapositivas.
\end{itemize}

\begin{ejemplo}[Energía cinética y velocidad]
Para una partícula con $K=mc^2$:
\begin{equation}
  \gamma=1+\frac{K}{mc^2}=2
  \Longrightarrow
  v=c\sqrt{1-\frac{1}{\gamma^2}}
  =\frac{\sqrt{3}}{2}\,c\approx0.866\,c.
  \label{eq:t5-ej-v}
\end{equation}
\end{ejemplo}
